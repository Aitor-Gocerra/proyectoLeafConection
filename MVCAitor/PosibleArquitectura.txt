│   │   └── Administrador.php   <-- CRUD para la tabla 'Administrador'
│   
│   └── /Views            <-- Interfaz de usuario (la 'V' del MVC)
│       ├── /layout       <-- Plantillas/layouts (header, footer, nav)

│       ├── /usuario      <-- Vistas relacionadas con el usuario

│       │   └── perfil.php
│       ├── /amigos       <-- Vistas de la red social
│       │   └── lista_amigos.php
│       ├── /juego        <-- Vistas de las partidas
│       │   ├── inicio_partida.php
│       │   └── resultado.php
│       └── /admin        <-- Vistas del panel de administración

├── /public               <-- Punto de entrada público (donde reside el index.php)
